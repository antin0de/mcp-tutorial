# Asynchronous Operations

Many tools need to perform long-running operations like API calls, file I/O, or database queries. This section covers proper async patterns in MCP servers.

## Async Request Handlers

All MCP request handlers are async functions. Always properly handle promises:

```typescript
server.setRequestHandler(CallToolRequestSchema, async (request) => {
  const { name, arguments: args } = request.params;

  if (name === 'search') {
    // This async operation is properly awaited
    const results = await database.query(args.query);
    return {
      content: [{
        type: 'text',
        text: JSON.stringify(results, null, 2),
      }],
    };
  }
});
```

## Timeouts

Set timeouts for external operations:

```typescript
async function fetchWithTimeout(url: string, timeoutMs: number) {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeoutMs);

  try {
    const response = await fetch(url, {
      signal: controller.signal,
    });
    clearTimeout(timeoutId);
    return response;
  } catch (error) {
    clearTimeout(timeoutId);
    if (error.name === 'AbortError') {
      throw new Error(`Request timeout after ${timeoutMs}ms`);
    }
    throw error;
  }
}

server.setRequestHandler(CallToolRequestSchema, async (request) => {
  const { name, arguments: args } = request.params;

  if (name === 'fetch') {
    const { url } = args as { url: string };
    const response = await fetchWithTimeout(url, 30000);
    // ...
  }
});
```

## Parallel Operations

Run independent operations in parallel:

```typescript
server.setRequestHandler(CallToolRequestSchema, async (request) => {
  const { name, arguments: args } = request.params;

  if (name === 'compare') {
    const { url1, url2 } = args as { url1: string; url2: string };

    // Run fetches in parallel
    const [data1, data2] = await Promise.all([
      fetch(url1).then(r => r.json()),
      fetch(url2).then(r => r.json()),
    ]);

    return {
      content: [{
        type: 'text',
        text: JSON.stringify({ data1, data2 }, null, 2),
      }],
    };
  }
});
```

## Streaming Responses (Advanced)

For long-running operations, consider streaming updates:

```typescript
server.setRequestHandler(CallToolRequestSchema, async (request) => {
  const { name, arguments: args } = request.params;

  if (name === 'process_large_file') {
    const { filePath } = args as { filePath: string };

    // Process in chunks
    const results = [];
    for await (const chunk of processFileByChunk(filePath)) {
      results.push(chunk);
      // Could emit progress here
    }

    return {
      content: [{
        type: 'text',
        text: JSON.stringify(results, null, 2),
      }],
    };
  }
});
```

## Error Handling in Async Operations

Always handle errors in async operations:

```typescript
server.setRequestHandler(CallToolRequestSchema, async (request) => {
  const { name, arguments: args } = request.params;

  if (name === 'fetch') {
    try {
      const { url } = args as { url: string };
      const response = await fetch(url);
      const data = await response.json();
      return {
        content: [{
          type: 'text',
          text: JSON.stringify(data, null, 2),
        }],
      };
    } catch (error) {
      if (error instanceof TypeError && error.message.includes('fetch failed')) {
        throw new Error(`Network error: Could not reach ${url}`);
      }
      throw error;
    }
  }
});
```

## Cleanup

Ensure cleanup happens even on errors:

```typescript
server.setRequestHandler(CallToolRequestSchema, async (request) => {
  const { name, arguments: args } = request.params;

  if (name === 'process') {
    const tempFile = await createTempFile();

    try {
      const result = await processFile(tempFile);
      return {
        content: [{
          type: 'text',
          text: result,
        }],
      };
    } finally {
      // Always cleanup
      await deleteTempFile(tempFile);
    }
  }
});
```
