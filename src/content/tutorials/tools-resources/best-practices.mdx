# Best Practices

Building production-ready MCP tools requires attention to error handling, security, and established patterns. This section covers the essential best practices that will make your tools robust, secure, and maintainable.

## Error Handling Patterns

### Structured Error Handling

```typescript
import { ToolError } from '@modelcontextprotocol/sdk/sdk';

  constructor(message: string, resourceName: string) {

  }

    throw new ToolValidationError('Input too long (max 1000 characters)');
  }
```

### Error Recovery Patterns

```typescript
import { z } from 'zod';

class ResilientCalculator {
  private maxRetries = 3;
  private retryDelay = 1000;

  async calculateWithRetry(
    operation: string,
    a: number,
    b: number
  ): Promise<number> {
    let lastError: Error;

    for (let attempt = 1; attempt <= this.maxRetries; attempt++) {
      try {
        return await this.performCalculation(operation, a, b);
      } catch (error) {
        lastError = error as Error;

        // Don't retry on certain errors
        if (this.isNonRetryableError(error)) {
          break;
        }

        // Wait before retrying
        if (attempt < this.maxRetries) {
          await new Promise(resolve => setTimeout(resolve, this.retryDelay * attempt));
        }
      }
    }

    throw new ToolError(
      `Calculation failed after ${this.maxRetries} attempts: ${lastError?.message}`,
      'CALCULATION_FAILED',
      lastError
    );
  }

  private async performCalculation(
    operation: string,
    a: number,
    b: number
  ): Promise<number> {
    // Perform the calculation
    switch (operation) {
      case 'add':
        return a + b;
      case 'subtract':
        return a - b;
      case 'multiply':
        return a * b;
      case 'divide':
        if (b === 0) {
          throw new ToolValidationError('Cannot divide by zero');
        }
        return a / b;
      default:
        throw new ToolValidationError('Invalid operation');
    }
  }

  private isNonRetryableError(error: any): boolean {
    return error instanceof ToolValidationError ||
           error instanceof ToolSecurityError;
  }
}
```

### Circuit Breaker Pattern

```typescript
class CircuitBreaker {
  private failureCount = 0;
  private failureThreshold = 5;
  private cooldownPeriod = 60000; // 1 minute
  private lastFailureTime = 0;
  private state: 'closed' | 'open' | 'half-open' = 'closed';

  async execute<T>(operation: () => Promise<T>): Promise<T> {
    if (this.state === 'open') {
      if (Date.now() - this.lastFailureTime < this.cooldownPeriod) {
        throw new ToolError('Circuit breaker is open', 'CIRCUIT_BREAKER_OPEN');
      } else {
        this.state = 'half-open';
      }
    }

    try {
      const result = await operation();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  private onSuccess(): void {
    this.failureCount = 0;
    this.state = 'closed';
  }

  private onFailure(): void {
    this.failureCount++;
    this.lastFailureTime = Date.now();

    if (this.failureCount >= this.failureThreshold) {
      this.state = 'open';
    }
  }
}

// Usage example
const circuitBreaker = new CircuitBreaker();

export const reliableFileOperation = async (filePath: string, operation: string) => {
  return circuitBreaker.execute(async () => {
    switch (operation) {
      case 'read':
        return fs.readFile(filePath, 'utf-8');
      case 'write':
        return fs.writeFile(filePath, 'content', 'utf-8');
      default:
        throw new ToolValidationError('Invalid operation');
    }
  });
};
```

## Security Best Practices

### Input Sanitization

```typescript
import { z } from 'zod';

class SecurityUtils {
  // Sanitize file paths to prevent directory traversal
  static sanitizePath(path: string): string {
    // Remove null bytes
    path = path.replace(/\x00/g, '');

    // Normalize path
    path = path.normalize();

    // Check for directory traversal
    if (path.includes('..')) {
      throw new ToolSecurityError('Directory traversal not allowed');
    }

    return path;
  }

  // Sanitize strings for safe output
  static sanitizeString(input: string, maxLength: number = 1000): string {
    // Remove null bytes
    input = input.replace(/\x00/g, '');

    // Truncate to maximum length
    if (input.length > maxLength) {
      input = input.substring(0, maxLength);
    }

    // Escape HTML if needed
    input = this.escapeHtml(input);

    return input;
  }

  // Escape HTML to prevent XSS
  private static escapeHtml(text: string): string {
    const map: Record<string, string> = {
      '&': '&amp;',
      '<': '&lt;',
      '>': '&gt;',
      '"': '&quot;',
      "'": '&#039;'
    };
    return text.replace(/[&<>"']/g, m => map[m]);
  }

  // Validate email format
  static isValidEmail(email: string): boolean {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email);
  }

  // Validate URL format
  static isValidUrl(url: string): boolean {
    try {
      new URL(url);
      return true;
    } catch {
      return false;
    }
  }
}

// Schema with security validation
const secureUserSchema = z.object({
  username: z.string()
    .min(3)
    .max(20)
    .regex(/^[a-zA-Z0-9_]+$/, 'Username can only contain letters, numbers, and underscores'),

  email: z.string()
    .refine(SecurityUtils.isValidEmail, 'Invalid email format'),

  bio: z.string()
    .max(500)
    .transform(val => SecurityUtils.sanitizeString(val)),
});

export { SecurityUtils, secureUserSchema };
```

### Rate Limiting

```typescript
import { ToolError } from '@modelcontextprotocol/sdk/sdk';

class RateLimiter {
  private requests = new Map<string, { count: number; resetTime: number }>();
  private windowMs = 60000; // 1 minute
  private maxRequests = 100;

  async checkRateLimit(userId: string): Promise<void> {
    const now = Date.now();
    const userRequest = this.requests.get(userId);

    // Clean up old entries
    if (userRequest && now > userRequest.resetTime) {
      this.requests.delete(userId);
    }

    const currentRequest = this.requests.get(userId) || { count: 0, resetTime: now + this.windowMs };

    if (currentRequest.count >= this.maxRequests) {
      throw new ToolError(
        `Rate limit exceeded. Try again after ${Math.ceil((currentRequest.resetTime - now) / 1000)} seconds`,
        'RATE_LIMIT_EXCEEDED'
      );
    }

    currentRequest.count++;
    currentRequest.resetTime = now + this.windowMs;
    this.requests.set(userId, currentRequest);
  }

  resetRateLimit(userId: string): void {
    this.requests.delete(userId);
  }
}

// Usage example
const rateLimiter = new RateLimiter();

export const rateLimitedTool = {
  name: 'sensitiveOperation',
  description: 'Operation with rate limiting',
  inputSchema: z.object({
    data: z.string(),
  }),
  handler: async (params: { data: string }) => {
    const userId = 'user-123'; // In real app, get from authentication

    await rateLimiter.checkRateLimit(userId);

    // Perform the operation
    return { success: true, data: params.data };
  },
};
```

### Permission Management

```typescript
interface User {
  id: string;
  role: 'admin' | 'user' | 'guest';
  permissions: string[];
}

class PermissionManager {
  private permissions = new Map<string, Set<string>>();

  constructor() {
    this.setupDefaultPermissions();
  }

  private setupDefaultPermissions(): void {
    this.permissions.set('admin', new Set(['*'])); // All permissions
    this.permissions.set('user', new Set([
      'read:files',
      'write:files',
      'execute:tools'
    ]));
    this.permissions.set('guest', new Set([
      'read:files',
      'read:public'
    ]));
  }

  checkPermission(user: User, resource: string, action: string): boolean {
    const userPermissions = this.permissions.get(user.role) || new Set();

    // Admin has all permissions
    if (userPermissions.has('*')) {
      return true;
    }

    const permission = `${action}:${resource}`;

    // Check direct permission
    if (userPermissions.has(permission)) {
      return true;
    }

    // Check wildcard permissions
    for (const userPerm of userPermissions) {
      if (userPerm.includes('*')) {
        const pattern = userPerm.replace('*', '.*');
        const regex = new RegExp(`^${pattern}$`);
        if (regex.test(permission)) {
          return true;
        }
      }
    }

    return false;
  }
}

// Usage example
const permissionManager = new PermissionManager();

export const checkUserPermission = (user: User, resource: string, action: string) => {
  if (!permissionManager.checkPermission(user, resource, action)) {
    throw new ToolError(
      `Access denied: ${action} on ${resource}`,
      'PERMISSION_DENIED'
    );
  }
};
```

## Performance Optimization

### Caching

```typescript
import { ToolError } from '@modelcontextprotocol/sdk/sdk';

class CacheManager {
  private cache = new Map<string, { data: any; timestamp: number; ttl: number }>();
  private maxSize = 1000;
  private cleanupInterval = 300000; // 5 minutes

  constructor() {
    this.startCleanup();
  }

  get<T>(key: string): T | null {
    const item = this.cache.get(key);

    if (!item) return null;

    // Check if expired
    if (Date.now() - item.timestamp > item.ttl) {
      this.cache.delete(key);
      return null;
    }

    return item.data as T;
  }

  set<T>(key: string, data: T, ttl: number = 60000): void {
    // Check if cache is full
    if (this.cache.size >= this.maxSize) {
      this.evictOldest();
    }

    this.cache.set(key, {
      data,
      timestamp: Date.now(),
      ttl
    });
  }

  private evictOldest(): void {
    let oldestKey = '';
    let oldestTimestamp = Infinity;

    for (const [key, item] of this.cache) {
      if (item.timestamp < oldestTimestamp) {
        oldestTimestamp = item.timestamp;
        oldestKey = key;
      }
    }

    if (oldestKey) {
      this.cache.delete(oldestKey);
    }
  }

  private startCleanup(): void {
    setInterval(() => {
      const now = Date.now();
      for (const [key, item] of this.cache) {
        if (now - item.timestamp > item.ttl) {
          this.cache.delete(key);
        }
      }
    }, this.cleanupInterval);
  }
}

// Usage example
const cache = new CacheManager();

export const getCachedUserInfo = async (userId: string) => {
  // Check cache first
  const cached = cache.get(userId);
  if (cached) {
    return cached;
  }

  // Not in cache, fetch from database
  const user = await fetchUserFromDatabase(userId);

  // Cache the result
  cache.set(userId, user, 300000); // 5 minutes TTL

  return user;
};
```

### Batch Processing

```typescript
import { z } from 'zod';
import { ToolError } from '@modelcontextprotocol/sdk/sdk';

const batchSchema = z.object({
  items: z.array(z.string()),
  operation: z.enum(['process', 'validate', 'transform']),
});

class BatchProcessor {
  private batchSize = 50;
  private delay = 100; // ms between batches

  async processBatch(
    items: string[],
    operation: string,
    processor: (item: string) => Promise<any>
  ): Promise<Array<{ item: string; result?: any; error?: string }>> {
    const results: any[] = [];

    // Process items in batches
    for (let i = 0; i < items.length; i += this.batchSize) {
      const batch = items.slice(i, i + this.batchSize);

      // Process batch concurrently
      const batchPromises = batch.map(async (item) => {
        try {
          const result = await processor(item);
          return { item, result };
        } catch (error) {
          return { item, error: error instanceof Error ? error.message : 'Unknown error' };
        }
      });

      const batchResults = await Promise.all(batchPromises);
      results.push(...batchResults);

      // Add delay between batches to avoid overwhelming the system
      if (i + this.batchSize < items.length) {
        await new Promise(resolve => setTimeout(resolve, this.delay));
      }
    }

    return results;
  }
}

// Usage example
const batchProcessor = new BatchProcessor();

export const batchFileProcessor = {
  name: 'batchProcessFiles',
  description: 'Process multiple files in batches',
  inputSchema: batchSchema,
  handler: async (params: z.infer<typeof batchSchema>) => {
    const { items, operation } = params;

    if (items.length > 1000) {
      throw new ToolError('Maximum 1000 items allowed in a single request', 'TOO_MANY_ITEMS');
    }

    const processor = async (filePath: string) => {
      // Process the file
      const content = await fs.readFile(filePath, 'utf-8');
      return { size: content.length, lines: content.split('\n').length };
    };

    const results = await batchProcessor.processBatch(items, operation, processor);

    return {
      total: items.length,
      processed: results.filter(r => !r.error).length,
      failed: results.filter(r => r.error).length,
      results,
    };
  },
};
```

```typescript

describe('CalculatorService', () => {

  });
```

### Configuration Management

```typescript
import { z } from 'zod';

// Configuration schema
const configSchema = z.object({
  server: z.object({
    port: z.number().default(3000),
    host: z.string().default('localhost'),
  }),
  database: z.object({
    url: z.string().url(),
    poolSize: z.number().default(10),
  }),
  security: z.object({
    rateLimit: z.object({
      windowMs: z.number().default(60000),
      maxRequests: z.number().default(100),
    }),
    jwtSecret: z.string().min(32),
  }),
  tools: z.object({
    maxHistory: z.number().default(100),
    defaultTimeout: z.number().default(30000),
    enableDebug: z.boolean().default(false),
  }),
});

type Config = z.infer<typeof configSchema>;

class ConfigManager {
  private config: Config;

  constructor(config: Record<string, any>) {
    this.config = configSchema.parse(config);
  }

  get server() {
    return this.config.server;
  }

  get database() {
    return this.config.database;
  }

  get security() {
    return this.config.security;
  }

  get tools() {
    return this.config.tools;
  }

  get(path: string): any {
    return path.split('.').reduce((obj, key) => obj?.[key], this.config);
  }

  set(path: string, value: any): void {
    const keys = path.split('.');
    const lastKey = keys.pop()!;
    const target = keys.reduce((obj, key) => obj?.[key], this.config);
    if (target) {
      target[lastKey] = value;
    }
  }
}

// Load configuration
export const config = new ConfigManager({
  server: {
    port: process.env.PORT ? parseInt(process.env.PORT) : 3000,
  },
  database: {
    url: process.env.DATABASE_URL || 'mongodb://localhost:27017',
    poolSize: 10,
  },
  security: {
    rateLimit: {
      windowMs: 60000,
      maxRequests: 100,
    },
    jwtSecret: process.env.JWT_SECRET || 'your-secret-key',
  },
  tools: {
    maxHistory: 100,
    defaultTimeout: 30000,
    enableDebug: process.env.NODE_ENV === 'development',
  },
});
```
